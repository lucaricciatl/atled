function(copy_res_to_target target)
    add_custom_command(TARGET ${target} POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_directory
            "${CMAKE_SOURCE_DIR}/resources"         # Path to your res folder in the main project
            "$<TARGET_FILE_DIR:${target}>/resources"  # Destination: where the executable is located
        COMMENT "Copying res folder to $<TARGET_FILE_DIR:${target}>"
    )
endfunction()
