# Set minimum CMake version
cmake_minimum_required(VERSION 3.10)

# Project name
project(Shapes2D)

# Add the source files
set(SOURCES
    src/main.cpp
    src/Engine.cpp
)

add_executable(${PROJECT_NAME} ${SOURCES})

target_include_directories(Shapes2D PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/include)

target_link_libraries(${PROJECT_NAME} PUBLIC atled)

set_target_properties(Shapes2D PROPERTIES FOLDER "examples")

function(copy_res_to_target target)
    add_custom_command(TARGET ${target} POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_directory
            "${CMAKE_SOURCE_DIR}/resources"         # Path to your res folder in the main project
            "$<TARGET_FILE_DIR:${target}>/resources"  # Destination: where the executable is located
        COMMENT "Copying res folder to $<TARGET_FILE_DIR:${target}>"
    )
endfunction()


copy_res_to_target(Shapes2D)
