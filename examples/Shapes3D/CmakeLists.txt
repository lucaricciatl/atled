# Set minimum CMake version
cmake_minimum_required(VERSION 3.10)

# Project name
project(Shapes3D)

# Add the source files
set(SOURCES
    src/main.cpp
    src/Engine.cpp
)

add_executable(${PROJECT_NAME} ${SOURCES})

target_include_directories(Shapes3D PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/include)

target_link_libraries(${PROJECT_NAME} PUBLIC atled)

set_target_properties(Shapes3D PROPERTIES FOLDER "examples")
