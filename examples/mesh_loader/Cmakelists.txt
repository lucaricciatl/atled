# Set minimum CMake version
cmake_minimum_required(VERSION 3.10)

# Project name
project(mesh_loader)

# Add the source files
set(SOURCES
    src/main.cpp
)

add_executable(${PROJECT_NAME} ${SOURCES})

target_include_directories(mesh_loader PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/include)

target_link_libraries(${PROJECT_NAME} PUBLIC atled)

set_target_properties(mesh_loader PROPERTIES FOLDER "examples")
